\titledquestion{Sumas, Riemann y Propiedades}[20]
Evalúe las sumas y aplique propiedades (asuma $\int_0^5 f(x)dx=10, \int_5^7 f(x)dx=3, \int_0^5 g(x)dx=-2$).
\begin{multicols}{2}
\begin{parts}
\part $\displaystyle \sum_{k=1}^{20} (3k - 2)$ \begin{solution} $590$ \end{solution}
\part $\displaystyle \sum_{i=1}^{15} (i^2 - 4i + 1)$ \begin{solution} $775$ \end{solution}
\part $\displaystyle \lim_{n \to \infty} \sum_{i=1}^{n} (x_i^*)^3 \Delta x$, en el intervalo $[0, 2]$. \begin{solution} $4$ \end{solution}
\part $\displaystyle \lim_{n \to \infty} \sum_{i=1}^{n} e^{x_i^*} \Delta x$, en el intervalo $[0, 1]$. \begin{solution} $e - 1$ \end{solution}
\part $\displaystyle \sum_{i=1}^{n} \frac{1}{\sqrt{i+1} + \sqrt{i}}$ \begin{solution} $\sqrt{n+1} - 1$ \end{solution}
\part $\displaystyle \sum_{i=1}^{n} \frac{2}{4i^2 - 1}$ \begin{solution} $1 - \frac{1}{2n+1}$ \end{solution}
\part $\displaystyle \int_0^7 f(x) \, dx$ \begin{solution} $13$ \end{solution}
\part $\displaystyle \int_5^0 f(x) \, dx$ \begin{solution} $-10$ \end{solution}
\part $\displaystyle \int_0^5 [3f(x) - 4g(x)] \, dx$ \begin{solution} $38$ \end{solution}
\part $\displaystyle \int_5^5 [f(x) \cdot g(x)] \, dx$ \begin{solution} $0$ \end{solution}
\end{parts}
\end{multicols}

\end{parts}
