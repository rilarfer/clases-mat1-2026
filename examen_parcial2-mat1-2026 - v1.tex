% Problema 1
\question Calcule la longitud de arco de la parábola $y = \frac{1}{2}x^2$ en el intervalo $0 \le x \le 1$.

\begin{center}
\begin{tikzpicture}[scale=2]
    % Ejes
    \draw[->] (-0.2,0) -- (1.5,0) node[right] {$x$};
    \draw[->] (0,-0.2) -- (0,0.8) node[above] {$y$};
    % Curva
    \draw[domain=0:1, smooth, variable=\x, blue, thick] plot ({\x}, {0.5*\x*\x});
    % Líneas guía y etiquetas
    \draw[dashed] (1,0) node[below] {$1$} -- (1,0.5);
    \draw[dashed] (0,0.5) node[left] {$1/2$} -- (1,0.5);
\end{tikzpicture}
\end{center}

\begin{solution}
La fórmula de la longitud de arco es $L = \int_a^b \sqrt{1 + [f'(x)]^2} \, dx$. \\
Calculamos la derivada: $f'(x) = x$. \\
Planteamos la integral:
\[ L = \int_0^1 \sqrt{1 + x^2} \, dx \]
Para resolver, aplicamos la sustitución trigonométrica $x = \tan(\theta)$, donde $dx = \sec^2(\theta) \, d\theta$. \\
La integral se transforma en:
\[ \int \sqrt{1 + \tan^2(\theta)} \sec^2(\theta) \, d\theta = \int \sec^3(\theta) \, d\theta \]
Resolviendo por partes o utilizando la fórmula de reducción:
\[ \int \sec^3(\theta) \, d\theta = \frac{1}{2}\sec(\theta)\tan(\theta) + \frac{1}{2}\ln|\sec(\theta) + \tan(\theta)| \]
Regresando a la variable $x$:
\[ L = \left[ \frac{x\sqrt{1+x^2}}{2} + \frac{1}{2}\ln\left|x + \sqrt{1+x^2}\right| \right]_0^1 \]
Evaluando los límites:
\[ L = \left( \frac{\sqrt{2}}{2} + \frac{1}{2}\ln(1+\sqrt{2}) \right) - (0 + 0) = \frac{\sqrt{2} + \ln(1+\sqrt{2})}{2} \approx 1.1478 \]
\end{solution}

% Problema 2
\question Determine el volumen del sólido generado al rotar alrededor del eje $y$ la región en el primer cuadrante acotada por la curva $y = \cos(x^2)$ y los ejes coordenados.

\begin{center}
\begin{tikzpicture}[scale=2]
    % Ejes
    \draw[->] (-0.2,0) -- (1.8,0) node[right] {$x$};
    \draw[->] (0,-0.2) -- (0,1.5) node[above] {$y$};
    % Región sombreada
    \fill[blue!10] (0,0) -- plot[domain=0:1.2533, samples=100] ({\x}, {cos(\x*\x*180/pi)}) -- (1.2533,0) -- cycle;
    % Curva
    \draw[domain=0:1.2533, smooth, samples=100, variable=\x, blue, thick] plot ({\x}, {cos(\x*\x*180/pi)});
    % Etiquetas
    \draw[dashed] (1.2533,0) node[below] {$\sqrt{\pi/2}$} -- (1.2533,0);
    \node[left] at (0,1) {$1$};
\end{tikzpicture}
\end{center}

\begin{solution}
La región interseca el eje $x$ cuando $\cos(x^2) = 0$, lo cual en el primer cuadrante ocurre en $x^2 = \pi/2$, es decir, $x = \sqrt{\pi/2}$. \\
Utilizando el método de cascarones cilíndricos, el diferencial de volumen es $dV = 2\pi x f(x) \, dx$.
\[ V = \int_0^{\sqrt{\pi/2}} 2\pi x \cos(x^2) \, dx \]
Aplicamos el cambio de variable $u = x^2$, de donde $du = 2x \, dx$. \\
Los nuevos límites de integración son: \\
Si $x = 0 \implies u = 0$ \\
Si $x = \sqrt{\pi/2} \implies u = \pi/2$ \\
Sustituyendo en la integral:
\[ V = \pi \int_0^{\pi/2} \cos(u) \, du \]
Integrando:
\[ V = \pi \Big[ \sin(u) \Big]_0^{\pi/2} = \pi (\sin(\pi/2) - \sin(0)) = \pi(1 - 0) = \pi \]
\end{solution}

% Problema 3
\question Halle el área de la región comprendida entre la curva amortiguada $y = e^{-x}\sin(x)$ y el eje $x$, en el intervalo de $x = 0$ a $x = \pi$.

\begin{center}
\begin{tikzpicture}[scale=1.5]
    % Ejes
    \draw[->] (-0.2,0) -- (3.5,0) node[right] {$x$};
    \draw[->] (0,-0.2) -- (0,0.8) node[above] {$y$};
    % Región sombreada
    \fill[blue!10] plot[domain=0:pi, samples=100] ({\x}, {exp(-\x)*sin(\x*180/pi)}) -- (pi,0) -- (0,0);
    % Curva
    \draw[domain=0:pi, smooth, samples=100, variable=\x, blue, thick] plot ({\x}, {exp(-\x)*sin(\x*180/pi)});
    % Etiquetas
    \node[below] at (pi,0) {$\pi$};
\end{tikzpicture}
\end{center}

\begin{solution}
El área está dada por la integral definida:
\[ A = \int_0^\pi e^{-x}\sin(x) \, dx \]
Resolvemos la integral indefinida $\int e^{-x}\sin(x) \, dx$ usando integración por partes. \\
Sea $u = \sin(x) \implies du = \cos(x) \, dx$ \\
Sea $dv = e^{-x} \, dx \implies v = -e^{-x}$ \\
Aplicando la fórmula $\int u \, dv = uv - \int v \, du$:
\[ \int e^{-x}\sin(x) \, dx = -e^{-x}\sin(x) + \int e^{-x}\cos(x) \, dx \]
Aplicamos partes nuevamente a la integral restante. \\
Sea $u = \cos(x) \implies du = -\sin(x) \, dx$ \\
Sea $dv = e^{-x} \, dx \implies v = -e^{-x}$
\[ \int e^{-x}\sin(x) \, dx = -e^{-x}\sin(x) - e^{-x}\cos(x) - \int e^{-x}\sin(x) \, dx \]
Agrupando términos semejantes:
\[ 2 \int e^{-x}\sin(x) \, dx = -e^{-x}(\sin(x) + \cos(x)) \]
\[ \int e^{-x}\sin(x) \, dx = -\frac{e^{-x}}{2}(\sin(x) + \cos(x)) \]
Evaluamos de $0$ a $\pi$:
\[ A = \left[ -\frac{e^{-\pi}}{2}(\sin(\pi) + \cos(\pi)) \right] - \left[ -\frac{e^{0}}{2}(\sin(0) + \cos(0)) \right] \]
Sabiendo que $\sin(\pi) = 0$, $\cos(\pi) = -1$, $\sin(0) = 0$ y $\cos(0) = 1$:
\[ A = \left( -\frac{e^{-\pi}}{2}(-1) \right) - \left( -\frac{1}{2}(1) \right) = \frac{e^{-\pi}}{2} + \frac{1}{2} = \frac{1 + e^{-\pi}}{2} \]
\end{solution}
